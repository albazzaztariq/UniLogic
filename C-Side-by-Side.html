Example 1 — Signal Handler Types
C
typedef void __signalfn_t(int);
typedef __signalfn_t __user *__sighandler_t;

typedef void __restorefn_t(void);
typedef __restorefn_t __user *__sigrestore_t;

#define SIG_DFL ((__sighandler_t)0)
#define SIG_IGN ((__sighandler_t)1)
Pragma
// No typedef needed — function pointer is just an address
none <handler> = sigHandler()    // signal handler function pointer
none <restorer> = sigRestorer()  // restore function pointer

constant none <SIG_DFL> = empty          // default: null pointer
constant none <SIG_IGN> = change(1)->none // ignore: sentinel value

Example 3 — GDT Pointer + Load
C
struct desc_ptr {
    unsigned short size;
    unsigned long address;
} __attribute__((packed));

static inline void native_load_gdt(
    const struct desc_ptr *dtr)
{
    asm volatile("lgdt %0"::"m" (*dtr));
}
Pragma
type desc_ptr
  uint16 size
  uint64 address
end type
pack(desc_ptr)

function load_gdt(desc_ptr <dtr>) returns none
  asm("lgdt %0" :: "m" (deref(<dtr>)))
end function
Example 4 — copy_pte_range (Linux kernel memory management)
C
/*
 * copy_pte_range - copy page table entries
 * from src to dst virtual area
 */
static int copy_pte_range(
    struct vm_area_struct *dst_vma,
    struct vm_area_struct *src_vma,
    pmd_t *dst_pmd,
    pmd_t *src_pmd,
    unsigned long addr,
    unsigned long end)
{
    struct mm_struct *dst_mm = dst_vma->vm_mm;
    struct mm_struct *src_mm = src_vma->vm_mm;
    pte_t *orig_src_pte, *orig_dst_pte;
    pte_t *src_pte, *dst_pte;
    pte_t ptent;
    spinlock_t *src_ptl, *dst_ptl;
    int progress, max_nr;
    int ret = 0;
    long rss[NR_MM_COUNTERS];
    swp_entry_t entry = (swp_entry_t){0};
    struct folio *prealloc = NULL;
    int nr;

again:
    progress = 0;
    init_rss_vec(rss);

    src_pte = pte_offset_map_lock(
        src_mm, src_pmd, addr, &src_ptl);
    if (!src_pte) {
        pmd_populate(dst_mm, dst_pmd,
            pmd_pgtable(*dst_pmd));
        ret = 0;
        goto out;
    }
    dst_pte = pte_offset_map_nolock(
        dst_mm, dst_pmd, addr, &dst_ptl);
    if (!dst_pte) {
        pte_unmap_unlock(src_pte, src_ptl);
        ret = -ENOMEM;
        goto out;
    }
    orig_src_pte = src_pte;
    orig_dst_pte = dst_pte;
    arch_enter_lazy_mmu_mode();

    do {
        nr = 1;
        if (progress >= 32) {
            progress = 0;
            if (need_resched() ||
                spin_needbreak(src_ptl) ||
                spin_needbreak(dst_ptl))
                break;
        }
        ptent = ptep_get(src_pte);
        if (pte_none(ptent)) {
            progress++;
            continue;
        }
        if (!pte_present(ptent)) {
            ret = copy_nonpresent_pte(
                dst_mm, src_mm,
                dst_pte, src_pte,
                dst_vma, src_vma,
                addr, rss);
            if (ret == -EIO) {
                entry = pte_to_swp_entry(
                    ptep_get(src_pte));
                break;
            } else if (ret == -EBUSY) {
                break;
            } else if (ret) {
                ret = 0;
            }
            progress += 8;
            continue;
        }
        ret = copy_present_pte(
            dst_vma, src_vma,
            dst_pte, src_pte, ptent,
            addr, rss, &prealloc);
        if (ret == -EAGAIN)
            break;
        if (prealloc) {
            folio_put(prealloc);
            prealloc = NULL;
        }
        progress += 8;
        dst_pte++;
        src_pte++;
        addr += (nr << PAGE_SHIFT);
    } while (addr != end);

    /* ... arch_leave_lazy_mmu_mode,
       unlock, update_mmu_tlb_range ... */
out:
    return ret;
}

// copy_pte_range — copy page table entries from src to dst
function copy_pte_range(vm_area_struct <dst_vma>,
  vm_area_struct <src_vma>, pmd_t <dst_pmd>, pmd_t <src_pmd>,
  uint64 addr, uint64 end) returns int
  none <dst_mm> = dst_vma.vm_mm  none <src_mm> = src_vma.vm_mm
  none <orig_src_pte>  none <orig_dst_pte>  none <src_pte>
  none <dst_pte>  pte_t ptent  none <src_ptl>  none <dst_ptl>
  int progress  int ret = 0  int nr
  array int[NR_MM_COUNTERS] rss
  softleaf_t entry = softleaf_mk_none()  none <prealloc> = empty
  while true
    progress = 0  init_rss_vec(rss)
    src_pte = pte_offset_map_lock(src_mm, src_pmd,
      addr, address(<src_ptl>))
    if src_pte == empty
      pmd_populate(dst_mm, dst_pmd, pmd_pgtable(deref(<dst_pmd>)))
      ret = 0  escape
    end if

      addr, address(<dst_ptl>))
    if dst_pte == empty
      pte_unmap_unlock(src_pte, src_ptl)  ret = -ENOMEM  escape
    end if
    orig_src_pte = src_pte  orig_dst_pte = dst_pte
    arch_enter_lazy_mmu_mode()
    while true
      nr = 1
      if progress >= 32
        progress = 0
        if need_resched() or spin_needbreak(src_ptl)
           or spin_needbreak(dst_ptl)
          escape
        end if
      end if
      ptent = ptep_get(src_pte)
      if pte_none(ptent)  progress++  continue  end if
      if not pte_present(ptent)
        ret = copy_nonpresent_pte(dst_mm, src_mm, dst_pte, src_pte,
          dst_vma, src_vma, addr, rss)
        if ret == -EIO
          entry = pte_to_swp_entry(ptep_get(src_pte))  escape
        else if ret == -EBUSY  escape
        else if ret not 0  ret = 0
        end if
        progress += 8  continue
      end if
      ret = copy_present_pte(dst_vma, src_vma, dst_pte, src_pte,
        ptent, addr, rss, address(<prealloc>))
      if ret == -EAGAIN  escape  end if
      if prealloc not empty  folio_put(prealloc)  prealloc = empty  end if
      progress += 8  dst_pte++  src_pte++
      addr += (nr left PAGE_SHIFT)
      if addr == end  escape  end if
    end while
    // out: arch_leave_lazy_mmu_mode,
    // unlock, update_mmu_tlb_range...
  end while
  return ret
end function

  vm_area_struct <src_vma>,
  pmd_t <dst_pmd>,
  pmd_t <src_pmd>,
  uint64 addr,
  uint64 end
) returns int

  none <src_mm> = src_vma.vm_mm
  none <orig_src_pte>
  none <orig_dst_pte>
  none <src_pte>
  none <dst_pte>
  pte_t ptent
  none <src_ptl>
  none <dst_ptl>
  int progress
  int max_nr
  int ret = 0

  none <prealloc> = empty
  int nr

    progress = 0
    init_rss_vec(rss)

    src_pte = pte_offset_map_lock(
      src_mm, src_pmd, addr, address(<src_ptl>))
    if src_pte == empty

        pmd_pgtable(deref(<dst_pmd>)))
      ret = 0
      escape  // goto out
    end if

      dst_mm, dst_pmd, addr, address(<dst_ptl>))
    if dst_pte == empty
      pte_unmap_unlock(src_pte, src_ptl)
      ret = -ENOMEM
      escape  // goto out
    end if

    orig_src_pte = src_pte
    orig_dst_pte = dst_pte

    while true
      nr = 1
      if progress >= 32
        progress = 0

           spin_needbreak(src_ptl) or
           spin_needbreak(dst_ptl)
          escape
        end if
      end if

      if pte_none(ptent)
        progress++
        continue
      end if

          dst_mm, src_mm,
          dst_pte, src_pte,
          dst_vma, src_vma,
          addr, rss)

            ptep_get(src_pte))

        else if ret not 0
          ret = 0
        end if

        continue

        dst_vma, src_vma,
        dst_pte, src_pte, ptent,
        addr, rss, address(<prealloc>))

        escape
      end if

        folio_put(prealloc)
        prealloc = empty
      end if

      src_pte++

    // unlock, update_mmu_tlb_range...
  end while

end function
goto out becomes escape from the outer while true. Pragma has no goto — the outer loop runs until an escape exits it. The inner do-while becomes a while true with an explicit if addr == end / escape / end if at the bottom.
