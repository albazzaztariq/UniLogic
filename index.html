#define container_of(ptr, type, member) ({ \
    void *__mptr = (void *)(ptr); \
    static_assert( \
        __same_type(*(ptr), \
            ((type *)0)->member) \
        || __same_type(*(ptr), void), \
        "pointer type mismatch"); \
    ((type *)(__mptr - \
        offsetof(type, member))); })
#function container_of(none <ptr>, uint64 off)
  none <__mptr> = change(<ptr>)->none
  uint64 addr = change(__mptr)->uint64 - off
  return change(addr)->none
#end function

class HttpRequest {
  private String url;
  private String method;
  private int timeout;
  private boolean followRedirects;
  private int retries;
  private String authToken;

  HttpRequest(String url, String method,
    int timeout, boolean followRedirects,
    int retries, String authToken) {
    this.url = url;
    this.method = method;
    this.timeout = timeout;
    this.followRedirects = followRedirects;
    this.retries = retries;
    this.authToken = authToken;
  }
}
type HttpRequest
  string url, method
  int timeout, followRedirects, retries
  string authToken
end type

function new_request(string url,
  string method, int timeout = 30,
  bool followRedirects = true,
  int retries = 3,
  string authToken = empty)
  return HttpRequest(url, method,
    timeout, followRedirects,
    retries, authToken)
end function

JavaScript — TimerUL
function Timer() {
  this.count = 0;
  var self = this;
  setInterval(function() {
      self.count++;
      if (self.count > 10) {
          self.stop();
      }
  }, 1000);
  this.stop = function() {
      var self = this;
      clearInterval(function() {
          self.count = 0;
      });
  };
}
type Timer
  int count
end type

function tick(Timer <t>)
  t.count++
  if t.count > 10
    stop(t)
  end if
end function

function stop(Timer <t>)
  t.count = 0
end function
