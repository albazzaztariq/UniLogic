The same pipeline — Python vs Pragma

Pragma describes a data pipeline as a flat sequence of named actions — fetch, convert, save, notify, and others. No boilerplate, no error handling scaffolding, no plumbing. The structure is the program.

Python
import requests
import json
import csv
import os

API_URL = "https://api.openalex.org/works"
params = {
    "filter": "concept.id:C41008148",
    "per-page": 100,
}
headers = {
    "Authorization": "Bearer mytoken123"
}

resp = requests.get(API_URL, params=params, headers=headers)
if resp.status_code != 200:
    raise Exception(
        f"Request failed: {resp.status_code}"
    )

data = resp.json()
records = data.get("results", [])

rows = []
for r in records:
    rows.append({
        "title": r.get("title", ""),
        "year":  r.get("publication_year", ""),
        "doi":   r.get("doi", ""),
    })

outpath = "output/records.csv"
os.makedirs("output", exist_ok=True)

with open(outpath, "w", newline="") as f:
    writer = csv.DictWriter(
        f, fieldnames=["title", "year", "doi"]
    )
    writer.writeheader()
    writer.writerows(rows)

print(f"Saved {len(rows)} records to {outpath}")
Pragma
// fetch from the API
fetch:
      from=[http or file]
      source=https://api.openalex.org/works
      auth=openalex_auth
      format=json
      structure=openalex.map
      name=records

// convert to csv
convert:
      content=records
      from=json
      to=csv
      name=report

// save to disk
save:
      to=file
      dest=output/records.csv
      content=report
      format=csv
      mode=overwrite
